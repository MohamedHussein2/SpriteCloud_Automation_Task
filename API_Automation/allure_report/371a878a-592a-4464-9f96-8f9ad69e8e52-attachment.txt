const jsonResponse = pm.response.json();

var orderId = pm.environment.get("orderId");

pm.test("Check code returns 200 in response", function () {
    pm.expect(jsonResponse.code).to.eql(200);
});

pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Check message returns deleted order", function () {
  pm.expect(jsonResponse.message).to.eql(orderId);
});

const jsonResponse = pm.response.json();

var orderId = pm.environment.get("orderId");

pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Check order id returns correct", function () {
  pm.expect(jsonResponse.id).to.eql(parseInt(orderId));
});

pm.test("Check that order status is correct", function () {
  pm.expect(jsonResponse.status).to.eql("placed");
});

pm.sendRequest("https://postman-echo.com/delay/1")
